Almost Union Find

Problem Statement

There are three operations you can do with this almost union-find data structure:

  • $1\;p\;q$: Union sets containing $p$ and $q$. Ignore if they are in the same set.
  • $2\;p\;q$: Move $p$ to set containing $q$. Ignore if they are in the same set.
  • $3\;p$: Return number of elements and sum of elements in set containing $p$.

Input

  • Integers $n$ and $m$, for number of integers and number of commands respectively.
  • $m$ lines with one operation per line.

Output

For each type-$3$ command, output 2 integers: number of elements and sum of elements.

Reference Code (Python)


import sys

def main():
    for line in sys.stdin:
        if not line.strip():
            continue
        n, m = map(int, line.split())

        max_nodes = n + m + 5  
        parent = list(range(max_nodes))
        cnt = [0] * max_nodes  
        tot = [0] * max_nodes 
        id = [0] * (n + 1)   
        next_node = n + 1

        for i in range(1, n + 1):
            id[i] = i
            parent[i] = i
            cnt[i] = 1
            tot[i] = i

        def find(x):
            while parent[x] != x:
                parent[x] = parent[parent[x]]
                x = parent[x]
            return x

        out = []
        for _ in range(m):
            parts = sys.stdin.readline().split()
            if not parts:
                continue
            t = int(parts[0])

            if t == 1:
                p, q = map(int, parts[1:3])
                rp, rq = find(id[p]), find(id[q])
                if rp != rq:
                    parent[rp] = rq
                    cnt[rq] += cnt[rp]
                    tot[rq] += tot[rp]

            elif t == 2:
                p, q = map(int, parts[1:3])
                rp, rq = find(id[p]), find(id[q])
                if rp != rq:
                    cnt[rp] -= 1
                    tot[rp] -= p
                    id[p] = next_node
                    parent[id[p]] = rq
                    cnt[rq] += 1
                    tot[rq] += p
                    next_node += 1

            else: 
                p = int(parts[1])
                r = find(id[p])
                out.append(f"{cnt[r]} {tot[r]}")

        sys.stdout.write("\n".join(out) + "\n")

if __name__ == "__main__":
    main()
